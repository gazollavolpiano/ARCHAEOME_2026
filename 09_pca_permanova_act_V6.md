# Ordination and PERMANOVA: bacteriome structure vs archaeal dominance groups

Tests if bacterial community composition differs across archaeal dominance groups

- 5% prevalence filter on species, then agglomerate to genus
- Bacteriome at species and genus level
- Pairwise and one-vs-rest PERMANOVA
- Adjust for age (or age_category), sex, BMI, read depth
- `betadisper` + `permutest` (dispersion vs location)

## PREDICT 1 and MetaCardis (BAKER)

Use  AsnicarF_2021 (1088 samples) and MetaCardis_2020_a (1726 samples). Both have sex, BMI and read depth; AsnicarF_2021 has continuous age, MetaCardis_2020_a only age_category

`physeq_list_...rds` was generated by `02_sample_selection_dedup.md`

`NPERM = 999` floors P at 1e-3, use 9999 for the final table!

run_pca <- function(X, meta, anchor = "M. sp900766745", contrast = "M. smithii") {
  pca <- prcomp(X, rank. = 5)          # only PC1/PC2 are plotted; sdev stays full-length
  var_pct <- 100 * pca$sdev^2 / sum(pca$sdev^2)
  scores  <- as.data.frame(pca$x)
  g       <- meta$Group

  if (mean(scores$PC1[g == anchor]) < mean(scores$PC1[g == contrast])) {
    scores$PC1            <- -scores$PC1
    pca$rotation[, "PC1"] <- -pca$rotation[, "PC1"]
    message("PC1 sign flipped for consistency.")
  }

  scores$SampleID <- rownames(scores)
  list(scores  = left_join(scores, rownames_to_column(meta, "SampleID"), by = "SampleID"),
       lab1    = sprintf("PC1 (%.1f%%)", var_pct[1]),
       lab2    = sprintf("PC2 (%.1f%%)", var_pct[2]))
}

# ---- 9. PERMANOVA -----------------------------------------------------------
permanova_row <- function(D, md, term, contrast, level) {
  vars    <- c(term, COVAR_NAMES)
  missing <- setdiff(vars, names(md))
  if (length(missing)) stop("Missing model variable(s): ", paste(missing, collapse = ", "))

  # adonis2 applies na.action to the whole data frame, not just formula variables.
  # Pass only what the model uses.
  md <- md[, vars, drop = FALSE]
  stopifnot(!anyNA(md))

  f <- if (FAST) as.formula(paste("D ~", COVARS, "+", term))
       else      as.formula(paste("D ~", term, "+", COVARS))
  by <- if (FAST) "terms" else "margin"

  message(sprintf("  [%s] %-30s n=%-5d ", level, contrast, nrow(md)), appendLF = FALSE)
  t0 <- Sys.time()
  a  <- adonis2(f, data = md, by = by, permutations = NPERM, parallel = NCORES)
  el <- as.numeric(difftime(Sys.time(), t0, units = "secs"))
  message(sprintf("R2=%.4f P=%.4g (%.1fs)", a[term, "R2"], a[term, "Pr(>F)"], el))

  tibble(Cohort = COHORT, Level = level, Transform = TRANSFORM, Contrast = contrast,
         R2 = a[term, "R2"], F = a[term, "F"], P = a[term, "Pr(>F)"],
         n = nrow(md), n_perm = NPERM, n_cores = NCORES, by = by, secs = el)
}

run_all_permanova <- function(M, meta, level) {
  D      <- as.dist(M)
  pairs  <- combn(levels(meta$Group), 2, simplify = FALSE)
  n_pair <- sum(map_lgl(pairs, ~ min(table(droplevels(meta$Group[meta$Group %in% .x]))) >= MIN_N))
  n_ovr  <- sum(table(meta$Group) >= MIN_N)
  message(sprintf("[%s] %d tests: 1 overall + %d pairwise + %d one-vs-rest",
                  level, 1 + n_pair + n_ovr, n_pair, n_ovr))

  overall <- permanova_row(D, meta, "Group", "Overall (4 groups)", level) %>%
    mutate(Family = "overall")

  pairwise <- map_dfr(pairs, function(p) {
    keep <- rownames(meta)[meta$Group %in% p]
    md   <- meta[keep, ] %>% mutate(Group = droplevels(Group))
    if (min(table(md$Group)) < MIN_N) {
      message(sprintf("  [%s] %-30s SKIPPED (n < %d)", level, paste(p, collapse = " vs "), MIN_N))
      return(NULL)
    }
    permanova_row(subset_dist(M, keep), md, "Group", paste(p, collapse = " vs "), level)
  }) %>% mutate(Family = "pairwise")

  ovr <- map_dfr(levels(meta$Group), function(g) {
    if (sum(meta$Group == g) < MIN_N) return(NULL)
    md <- meta %>% mutate(OVR = factor(ifelse(Group == g, "Focal", "Rest"),
                                       levels = c("Rest", "Focal")))
    permanova_row(D, md, "OVR", paste0(g, " vs rest"), level)
  }) %>% mutate(Family = "one_vs_rest")

  bind_rows(overall, pairwise, ovr)
}

# ---- 10. betadisper -----------------------------------------------------------
# betadisper cannot adjust for covariates: dispersion is tested unadjusted while location is tested adjusted 
run_betadisper <- function(D, meta, level) {
  bd <- betadisper(D, meta$Group, type = "median")   # matches by POSITION, not name
  pt <- permutest(bd, permutations = NPERM, pairwise = TRUE)
  list(
    omnibus = tibble(Cohort = COHORT, Level = level, Test = "betadisper (omnibus)",
                     F = pt$tab[1, "F"], P = pt$tab[1, "Pr(>F)"], n = nrow(meta)),
    pairwise = tibble(Cohort = COHORT, Level = level,
                      Contrast = names(pt$pairwise$permuted),
                      P = as.numeric(pt$pairwise$permuted)),
    spread = tibble(Cohort = COHORT, Level = level,
                    Group = names(tapply(bd$distances, bd$group, mean)),
                    mean_dist_to_median = as.numeric(tapply(bd$distances, bd$group, mean)),
                    n = as.integer(table(bd$group)))
  )
}

stopifnot(identical(rownames(M_genus), rownames(meta)))   # positional match is safe

run_pca <- function(X, meta, anchor = "M. sp900766745", contrast = "M. smithii") {
  pca <- prcomp(X, rank. = 5)          # only PC1/PC2 are plotted; sdev stays full-length
  var_pct <- 100 * pca$sdev^2 / sum(pca$sdev^2)
  scores  <- as.data.frame(pca$x)
  g       <- meta$Group
  
  if (mean(scores$PC1[g == anchor]) < mean(scores$PC1[g == contrast])) {
    scores$PC1            <- -scores$PC1
    pca$rotation[, "PC1"] <- -pca$rotation[, "PC1"]
    message("PC1 sign flipped for consistency.")
  }
  
  scores$SampleID <- rownames(scores)
  list(scores  = left_join(scores, rownames_to_column(meta, "SampleID"), by = "SampleID"),
       lab1    = sprintf("PC1 (%.1f%%)", var_pct[1]),
       lab2    = sprintf("PC2 (%.1f%%)", var_pct[2]))
}

# ---- 9. PERMANOVA -----------------------------------------------------------
permanova_row <- function(D, md, term, contrast, level) {
  vars    <- c(term, COVAR_NAMES)
  missing <- setdiff(vars, names(md))
  if (length(missing)) stop("Missing model variable(s): ", paste(missing, collapse = ", "))
  
  # adonis2 applies na.action to the whole data frame, not just formula variables.
  # Pass only what the model uses.
  md <- md[, vars, drop = FALSE]
  stopifnot(!anyNA(md))
  
  f <- if (FAST) as.formula(paste("D ~", COVARS, "+", term))
  else      as.formula(paste("D ~", term, "+", COVARS))
  by <- if (FAST) "terms" else "margin"
  
  message(sprintf("  [%s] %-30s n=%-5d ", level, contrast, nrow(md)), appendLF = FALSE)
  t0 <- Sys.time()
  a  <- adonis2(f, data = md, by = by, permutations = NPERM, parallel = NCORES)
  el <- as.numeric(difftime(Sys.time(), t0, units = "secs"))
  message(sprintf("R2=%.4f P=%.4g (%.1fs)", a[term, "R2"], a[term, "Pr(>F)"], el))
  
  tibble(Cohort = COHORT, Level = level, Transform = TRANSFORM, Contrast = contrast,
         R2 = a[term, "R2"], F = a[term, "F"], P = a[term, "Pr(>F)"],
         n = nrow(md), n_perm = NPERM, n_cores = NCORES, by = by, secs = el)
}

run_all_permanova <- function(M, meta, level) {
  D      <- as.dist(M)
  pairs  <- combn(levels(meta$Group), 2, simplify = FALSE)
  n_pair <- sum(map_lgl(pairs, ~ min(table(droplevels(meta$Group[meta$Group %in% .x]))) >= MIN_N))
  n_ovr  <- sum(table(meta$Group) >= MIN_N)
  message(sprintf("[%s] %d tests: 1 overall + %d pairwise + %d one-vs-rest",
                  level, 1 + n_pair + n_ovr, n_pair, n_ovr))
  
  overall <- permanova_row(D, meta, "Group", "Overall (4 groups)", level) %>%
    mutate(Family = "overall")
  
  pairwise <- map_dfr(pairs, function(p) {
    keep <- rownames(meta)[meta$Group %in% p]
    md   <- meta[keep, ] %>% mutate(Group = droplevels(Group))
    if (min(table(md$Group)) < MIN_N) {
      message(sprintf("  [%s] %-30s SKIPPED (n < %d)", level, paste(p, collapse = " vs "), MIN_N))
      return(NULL)
    }
    permanova_row(subset_dist(M, keep), md, "Group", paste(p, collapse = " vs "), level)
  }) %>% mutate(Family = "pairwise")
  
  ovr <- map_dfr(levels(meta$Group), function(g) {
    if (sum(meta$Group == g) < MIN_N) return(NULL)
    md <- meta %>% mutate(OVR = factor(ifelse(Group == g, "Focal", "Rest"),
                                       levels = c("Rest", "Focal")))
    permanova_row(D, md, "OVR", paste0(g, " vs rest"), level)
  }) %>% mutate(Family = "one_vs_rest")
  
  bind_rows(overall, pairwise, ovr)
}

  run_all_permanova(M_species, meta, "Species")

# ---- 10. betadisper -----------------------------------------------------------
# betadisper cannot adjust for covariates: dispersion is tested unadjusted while location is tested adjusted 
run_betadisper <- function(D, meta, level) {
  bd <- betadisper(D, meta$Group, type = "median")   # matches by POSITION, not name
  pt <- permutest(bd, permutations = NPERM, pairwise = TRUE)
  list(
    omnibus = tibble(Cohort = COHORT, Level = level, Test = "betadisper (omnibus)",
                     F = pt$tab[1, "F"], P = pt$tab[1, "Pr(>F)"], n = nrow(meta)),
    pairwise = tibble(Cohort = COHORT, Level = level,
                      Contrast = names(pt$pairwise$permuted),
                      P = as.numeric(pt$pairwise$permuted)),
    spread = tibble(Cohort = COHORT, Level = level,
                    Group = names(tapply(bd$distances, bd$group, mean)),
                    mean_dist_to_median = as.numeric(tapply(bd$distances, bd$group, mean)),
                    n = as.integer(table(bd$group)))
  )
}
