# Archaeal detection per sample: cohort-level prevalence summaries and mixed-effects logistic regression of archaeal detection

The file `physeq_list_metagenomes_with_metadata_and_single_sample_per_individual.rds` was generated by `02_sample_selection_dedup.md`.

# load packages
library(tidyverse)
library(phyloseq)
library(microbiome)

# read new taxonomy data
physeq_list <- readRDS("physeq_list_metagenomes_with_metadata_and_single_sample_per_individual.rds")

# List the phyloseq objects to be tested 
studies <- c("AsnicarF_2021-1088samples", "JieZ_2017-385samples", "KarlssonFH_2013-145samples", "LiJ_2017-196samples", 
             "MetaCardis_2020_a-1726samples", "QinN_2014-237samples", "SchirmerM_2016-471samples",
             "YachidaS_2019-609samples", "ShaoY_2019-753samples", "MehtaRS_2018-308samples",
             "VatanenT_2016-212samples", "BackhedF_2015-200samples", "HMP_2019_ibdmdb-130samples", "BrooksB_2017-30samples")

# Read FR02 data
FR02_data <- read.csv("FR02_archaea_prevalence_for_age_tests.csv")

#######################################################################################################################
#      Calculate the prevalence of Archaea in the data for all individuals and by age group (adult, child, newborn, schoolage, senior etc.)
#   Age category: newborn < 1 year; 1 <= child < 12; 12 <= schoolage <19; 19 <= adult <= 65; senior > 65
#######################################################################################################################

# Initialize a dataframe to store prevalence results for all studies
Archaea_prevalence <- data.frame()
for(study_name in studies) {

  # Print the study name
  cat("Processing study:", study_name, "\n")
  
  # Filter for the study being analyzed
  phyloseq_obj <- physeq_list[[study_name]]

  # Melt the phyloseq object 
  psmelt <- phyloseq_obj %>% psmelt() 

  # List samples with archaea
  archaea_samples <- psmelt %>% filter(Domain== "Archaea" & Abundance > 0) %>% pull(SampleID) %>% unique()

  # Some studies have no age category information in the metadata
  # MehtaRS_2018 and LiJ_2017 --> all adults
  # KarlssonFH_2013           --> all seniors
  # BrooksB_2017              --> all newborns
  if (study_name == "MehtaRS_2018-308samples") {
    psmelt$age_category <- "adult"
  } else if (study_name == "LiJ_2017-196samples") {
    psmelt$age_category <- "adult"
  } else if (study_name == "KarlssonFH_2013-145samples") {
    psmelt$age_category <- "senior"
  } else if (study_name == "BrooksB_2017-30samples") {
    psmelt$age_category <- "newborn"
  } 

  # Organize metadata
  psmelt <- psmelt %>%
              as.data.frame() %>% 
              select(SampleID, age_category) %>%
              distinct() %>%
              mutate(Study = study_name)

  # Add if archaea is detected
  psmelt$Detection <- ifelse(psmelt$SampleID %in% archaea_samples, 1, 0)

  # Read data with read depth and add to psmelt
  file_path <- list.files(paste0("/labs/mericlab/curatedMetagenomicData/",strsplit(study_name, "-")[[1]][1]), pattern = "fastq_stats_", full.names = TRUE)
  fastq_stats <- read.table(file_path, header = FALSE, sep = "\t", stringsAsFactors = FALSE) %>% 
    rename(SampleID = V1, ReadDepth = V4) %>%
    mutate(SampleID = gsub(".*\\/(.*)_1.fastq", "\\1", SampleID)) %>%
    mutate(SampleID = gsub("-", ".", SampleID)) %>%
    filter(SampleID != "file") %>%
    mutate(ReadDepth = as.numeric(ReadDepth)) %>%
    select(SampleID, ReadDepth)

  psmelt <- psmelt %>% left_join(fastq_stats, by = "SampleID")
  
  # Add to Archaea_prevalence dataframe
  Archaea_prevalence <- rbind(Archaea_prevalence, psmelt)
}

library(lme4)
library(broom.mixed)

df <- Archaea_prevalence %>%
  filter(!is.na(ReadDepth), !is.na(age_category)) %>%
  mutate(log10_depth = log10(ReadDepth),
         age_category = factor(age_category, levels = c("adult","schoolage","child","newborn","senior")))

m_null  <- glmer(Detection ~ 1 + (1|Study), df, binomial)
m_depth <- glmer(Detection ~ scale(log10_depth) + (1|Study), df, binomial, control = glmerControl(optimizer = "bobyqa"))
m_age   <- glmer(Detection ~ age_category + (1|Study), df, binomial)
m_full  <- glmer(Detection ~ scale(log10_depth) + age_category + (1|Study), df, binomial, control = glmerControl(optimizer = "bobyqa"))

  mutate(
    cohort_label = if_else(location == "", cohort, paste0(cohort, " (", location, ")")),
    stratum_n = case_when(
      age_group == "All ages" ~ total_n,
      prevalence > 0 ~ round(positive_n / (prevalence / 100)),
      TRUE ~ NA_real_
    ),
    age_group = factor(age_group, levels = age_order)
  )

  filter(age_group == "All ages") %>%
  arrange(stratum_n) %>%
  pull(cohort_label)
